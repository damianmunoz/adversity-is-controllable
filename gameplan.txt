  ---                                                                                                    
  Data Flow Between Steps                                                                                
                  
  Step 1  →  data/raw/**/*.jsonl
             (trades: price, qty, is_buyer_maker, trade_time_ms)                                         
             (depth_updates: first_update_id, final_update_id, bids, asks)                               
                                                                                                         
  Step 2  →  BookSnapshot (in-memory + Parquet)                                                          
             (ts_ms, last_update_id, top-20 bids/asks)                                                   
                                                                                                         
  Step 3  →  FeatureVector (in-memory + Parquet)
             (ofi_l1..l5, spread_bps, microprice, depth_imbalance, vol_1/5/30s)                          
                                                                                                         
  Step 4  →  KalmanState
             (x_hat=[market_pressure, regime], P=2x2 covariance)                                         
                                                                                                         
  Step 5  →  Action (WAIT / PASSIVE / AGGRESSIVE)
                                                                                                         
  Step 6  →  FillResult → scalar loss → policy weight update                                             
             (fill_price, slippage, adverse_move logged to execution_log/)
                                                                                                         
  ---             
  Key Interfaces (signatures that lock in the contracts)                                                 
                                                                                                         
  # book/order_book.py
  class OrderBook:                                                                                       
      def apply_snapshot(self, snapshot: dict) -> None
      def apply_depth_update(self, event: dict) -> bool  # False = gap                                   
      def best_bid(self) -> PriceLevel                                                                   
      def best_ask(self) -> PriceLevel                                                                   
      def levels(self, side: str, n: int) -> list[PriceLevel]                                            
                                                                                                         
  # features/feature_pipeline.py
  @dataclass                                                                                             
  class FeatureVector:
      ts_ms: int; symbol: str
      ofi_l1..l5: float; spread_bps: float; microprice: float                                            
      depth_imbalance: float; vol_1s: float; vol_5s: float; vol_30s: float
                                                                                                         
  # state/kalman_filter.py
  class MarketKalmanFilter:                                                                              
      def predict(self) -> KalmanState
      def update(self, obs: FeatureVector) -> KalmanState
                                                                                                         
  # policy/hedge.py
  class HedgePolicy:                                                                                     
      def select(self, context: PolicyContext) -> Action
      def update(self, action: Action, loss: float) -> None                                              
      def save(self, path: Path) -> None                                                                 
                                                                                                         
  # execution/loss.py                                                                                    
  def compute_loss(fill: FillResult, arrival_mid: float,
                   exit_mid: float, lambda_risk: float) -> float
